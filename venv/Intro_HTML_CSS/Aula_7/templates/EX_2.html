import time

def contagem_regressiva():
for i in range(10, 0, -1):
print(i)
time.sleep(1)
print("Olha o aviaozinho!")

contagem_regressiva()

def contagem_recursiva(numero):
if numero == 0:
print("Quem quer dinheiro.")
return
print(numero)
contagem_recursiva(numero - 1)

contagem_recursiva(10)
